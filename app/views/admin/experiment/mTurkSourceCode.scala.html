<script>
	// Hint: What's called an experiment in MA is called a survey in MT
	var host = "@hostname";
	var experimentPath = "/ma/@experiment.id/start";
	var query = document.URL.split('?')[1];

	window.onload = function() {
		if ((typeof query === "undefined") || query.indexOf("ASSIGNMENT_ID_NOT_AVAILABLE") > -1) {
			// Show preview
			document.getElementById("previewHit").style.display = "block";
			document.getElementById("acceptedHit").style.display = "none";
			var experimentUrlHref = "";
			var experimentUrlText = "link will be shown when HIT is accepted";
		} else {
			// Show accepted HIT
			document.getElementById("previewHit").style.display = "none";
			document.getElementById("acceptedHit").style.display = "block";
			var experimentUrlHref = "http://" + host + experimentPath + "?" + query;
			var experimentUrlText = experimentUrlHref;
		}
		document.getElementById("experimentUrl").setAttribute("href", experimentUrlHref);
		document.getElementById("experimentUrl").innerHTML = experimentUrlText;
	}
</script>
